# push_back 
- add a element in back of the list 
- syntax 
    list_name.push_back(element) ;

# push_front
- add a element in front of the list 
- syntax 
    list_name.push_front(element) ;

# pop_back 
- delete a element at the end of the list 
- syntax 
    list_name.pop_back() ;

# pop_front
- delete a element from the front of the list
- syntax 
    list_name.pop_front() ;
